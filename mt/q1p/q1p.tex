\subsection{Section formula}
\begin{frame}
\frametitle{Section formula}
\solution{
Using Section Formula (\(k=3\)):
\begin{align}
	C &= \frac{kB+A}{k+1} \\
	C &= \frac{1}{3+1} \brak{3B + A} \\
    \implies	C &= \frac{1}{4}\brak{ \brak{3a + 3b} + \brak{2a - 3b}} \\
	C &= \frac{5}{4} a
\end{align}
}
\end{frame}
